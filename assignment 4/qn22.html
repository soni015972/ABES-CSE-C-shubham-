#include <bits/stdc++.h>
using namespace std;
int main() {
int m, n;
cin >> m >> n;
vector<vector<int>> matrix(m, vector<int>(n));
for (auto &row : matrix) {
for (auto &val : row) {
cin >> val;
}
}
for (int i = 0; i < m; i++) {
int sum = accumulate(matrix[i].begin(), matrix[i].end(), 0);
int mn = *min_element(matrix[i].begin(), matrix[i].end());
int mx = *max_element(matrix[i].begin(), matrix[i].end());
cout << sum << mx << mn << endl;
}
}
